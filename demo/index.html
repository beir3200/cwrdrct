This is demo from k-redirect component

Source Code : 

            
                <template>
                    <k-redirect mode="{{_setMode}}"></k-redirect>
                    <button on-click="_changeMode">set mode ON</button>
                </template>

                <script>
                    Polymer({
                        is: 'k-redirect-demo',
                        properties: 
                        {
                            _setMode: {
                                type: String,
                                value: "off",
                                notify: true
                            }
                        },
                        _changeMode: function(){
                            this.set("_setMode", "on");
                        }
                    });
                </script>
